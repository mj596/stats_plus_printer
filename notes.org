Example:
SELECT 
  d.AVG, d.STD, d.MIN, d.MAX
FROM PREDICTION_STATS s
LEFT JOIN PREDICTION_STATS_DATA d ON 
  s.ID_DATA = d.ID
WHERE
  s.CLIENT = "PEPCO" AND
  s.APPLICATION = "INT" AND
  s.TIME = 15 AND
  s.TYPE = "CUMSUM" AND
  s.TIMEBIN_AMOUNT = 1 AND s.TIMEBIN_UNIT = "H" AND
  s.FOLD_TIME_AMOUNT = 4 AND s.FOLD_TIME_UNIT = "W"
